```typescript title="WASM"
import init, { extractBytes } from "kreuzberg-wasm";
import {
  ChunkingConfig,
  ExtractionConfig,
  ChunkType,
} from "kreuzberg-wasm";

await init();

const fileBuffer = new Uint8Array(/* your file bytes */);
const mimeType = "application/pdf";

const config = new ExtractionConfig({
  chunking: new ChunkingConfig({
    enable_chunking: true,
    chunk_size: 512,
    chunk_overlap: 128,
    chunk_types: [ChunkType.Paragraph, ChunkType.Heading],
  }),
});

const result = await extractBytes(fileBuffer, mimeType, config);

if (result.chunks && result.chunks.length > 0) {
  console.log(`Generated ${result.chunks.length} chunks`);
  
  // Prepare chunks for vector database ingestion
  const vectorizedChunks = result.chunks.map((chunk, index) => {
    return {
      id: `chunk_${index}`,
      content: chunk.content,
      chunk_type: chunk.chunk_type,
      metadata: {
        page_number: chunk.metadata?.page_number,
        chunk_index: index,
        total_chunks: result.chunks.length,
        document_title: result.metadata?.title,
      },
      // Embedding would be generated by vector service
      embedding: chunk.embedding || undefined,
    };
  });
  
  console.log("Chunks prepared for vector database:");
  vectorizedChunks.forEach((chunk) => {
    console.log(
      `  [${chunk.id}] ${chunk.content.substring(0, 80)}...`
    );
    console.log(`    Metadata:`, chunk.metadata);
  });
  
  // If embeddings were generated, they would be available in chunk.embedding
  if (result.chunks[0].embedding) {
    console.log(
      `Embeddings available: ${result.chunks[0].embedding.length} dimensions`
    );
  }
} else {
  console.log("No chunks generated");
}
```
